from microbit import *
import speech 
import random 

# hier worden de worden gedifinieerd 
# 3 is de lengte van de arrays "onderwerp", "werkwoord" en "rest" 
#  als je meer woorden wilt gebruiken moet je de lengte veranderen 
lengteWoordArray = 3 

# Arrys met woorden
onderwerp = ["he", "Zarko", "she"] 
werkwoord = ["sleep", "eat", "run"] 
rest = ["outside", "inside", "at school"] 


def sayTheWords1(word): # function sayTheWords, argument word
    print(word) # laat in serial monitor de tekst zien
    speech.say(word, speed=100, pitch=230, throat=250, mouth=120)
    sleep(500) 


def sayTheWords2():
    willekeurigGetal = random.randint(0, lengteWoordArray-1) 
    display.show(willekeurigGetal) 
    sayTheWords1(onderwerp[willekeurigGetal])
    sayTheWords1(werkwoord[willekeurigGetal]) 
    sayTheWords1(rest[willekeurigGetal]) 
    
    
while True:
	if button_a.is_pressed():
            display.show(Image.SNAKE)
            sayTheWords1("dat was een slang") 
        elif button_b.is_pressed():
            display.show(Image.COW)
            sayTheWords1("dat was een koe") 
        elif display.read_light_level() < 40: 
            sayTheWords2()
        else:
            display.show(Image.SQUARE)
